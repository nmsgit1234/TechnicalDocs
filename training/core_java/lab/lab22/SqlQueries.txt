# Demonstration of the primary key

CREATE TABLE CUSTOMERS (
	CUSTOMERID INT UNIQUE NOT NULL, 
	CUSTOMERNAME VARCHAR(20),
	ADDRESS VARCHAR(50),
	CITY VARCHAR(20),
	POSTALCODE VARCHAR(10),
	COUNTRY VARCHAR(20),
	PRIMARY KEY (CUSTOMERID)
);
